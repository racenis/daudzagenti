66static List* parse_sexpr(const char** sexpr, short file_index, short* line) {
67 // let's start with abbreviating the parameter
68 const char* s = *sexpr;
69
70 // then we can create pointers for the list
71 List* list = NULL;
72 List* list_first = NULL;
73
74 // now we can start parsing in the atoms and values until we get to the end
75 // of the expression
76 while (1) {
77 // skip over any whitespace
78 while (isspace(*s)) *s++ == '\n' && (*line)++;
79
80 // skip over comment
81 if (*s == ';') {
82 while(*s != '\n' && *s != '\0') s++;
83 continue;
84 }
85
86 // if we're at the end of the s-expression, bail
87 if (*s == ')' || *s == '\0') {
88 break;
89 }
90
91 // now we can create a new list element
92 List* new_list = goetia_NewListElement();
93
94 new_list->file = file_index;
95 new_list->line = *line;
96
97 if (list) {
98 list->next = new_list;
99 }
100
101 if (!list_first) {
102 list_first = new_list;
103 }
104
105 list = new_list;
106
107 // check if number
108 if (isdigit(*s) || ((*s == '+' || *s == '-') && isdigit(*(s+1)))) {
109 list->data = goetia_value_MakeNumeric(strtof(s, NULL));
110 while (!isspace(*s) && *s != ')') s++;
111 } else switch (*s) {
112 // check if we have another s-expression
113 case '(': {
114 s++;
115 list->data.list = parse_sexpr(&s, file_index, line);
116 list->data.type = VALUE_LIST;
117 } break;
118
119 // check if we have a string
120 case '"': {
121 char buffer[100];
122 char* ptr = buffer;
123
124 // skip over first quote
125 s++;
126
127 // copy the string into buffer
128 while (*s != '"') {
129 // TODO: fix linebreak in string messing up line count
130 *(ptr++) = *(s++);
131 }
132
133 // skip over last quote and append a null to the buffer
134 *ptr = '\0';
135 s++;
136
137 // now we know how long the string is and we can allocate
138 char* value = malloc(ptr - buffer + 1);
139 strcpy(value, buffer);
140
141 list->data.str = value;
142 list->data.type = VALUE_STRING;
143 } break;
144
145 // otherwise we will assume that we have an atom
146 default: {
147 char buffer[100];
148 char* ptr = buffer;
149
150 while (!isspace(*s) && *s != ')') {
151 *(ptr++) = *(s++);
152 }
153
154 *ptr = '\0';
155
156 // now we know how long the string is and we can allocate
157 list->data.atom = goetia_atom_FromString(buffer);
158 list->data.type = VALUE_ATOM;
159 }
160 }
161 }
162
163 // skip over closing bracket, return parameter and return
164 s++;
165
166 *sexpr = s;
167
168 return list_first;
