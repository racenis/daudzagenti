53static Value inject(Context* c, List* p) {
54 if (!p || !p->next || !p->next->next) {
55 return error("Function 'inject' missing parameters.\n");
56 }
57
58 Agent* source_agent = goetia_world_FindAgent(find_self(c));
59 Agent* target_agent = goetia_world_FindAgent(goetia_script_EvaluateValue(c, p->data).int32);
60 if (!source_agent) {
61 return error("Function 'inject' _self %i not found.\n", find_self(c));
62 }
63 if (!target_agent) {
64 return error("Function 'inject' target %i not found.\n", p->data.int32);
65 }
66
67 chem chem = goetia_chemical_AtomToChemical(p->next->data.atom);
68 if (!chem) {
69 return error("Function 'inject' chemical %s not found.\n", goetia_atom_ToString(p->next->data.atom));
70 }
71
72 int ammount = p->next->next->data.int32;
73 if (ammount < 1) {
74 return error("Function 'inject' invalid chem ammount %i.\n", ammount);
75 }
76
77 ChemContainer* source = source_agent->chemicals;
78 ChemContainer* target = target_agent->chemicals;
79
80 if (!source) return ok;
81 if (!target) {
82 target_agent->chemicals = goetia_chemical_MakeContainer();
83 target = target_agent->chemicals;
84 }
85
86 int removed = goetia_chemical_Remove(source, chem, ammount);
87 goetia_chemical_Add(target, chem, removed);
88
89 printf("Injected %i of %i.\n", removed, chem);
90
91 return ok;
92}

376static Value lobe(Context* c, List* p) {
377 if (!p) {
378 return error("Function 'lobe' missing first parameter.\n");
379 }
380
381 Agent* agent = goetia_world_FindAgent(find_self(c));
382 if (!agent) {
383 return error("Function 'lobe' _self %i not found.\n", find_self(c));
384 }
385
386 if (p->data.type != VALUE_ATOM) {
387 return error("Function 'lobe' first parameter needs to be atom.\n");
388 }
389
390 switch (p->data.atom) {
391 case ATOM_MAKE:
392 if (!p->next || !p->next->next || !p->next->next->next) {
393 return error("Function 'lobe make' missing parameters.\n");
394 }
395
396 const List* layer_type = p->next;
397 const List* layer_name = p->next->next;
398
399 if (layer_type->data.type != VALUE_ATOM) {
400 return error("Function 'lobe make' layer type invalid.\n");
401 }
402
403 if (layer_name->data.type != VALUE_ATOM) {
404 return error("Function 'lobe make' layer name invalid.\n");
405 }
406
407 int layer_type_enum = 0;
408 switch (layer_type->data.atom) {
409 case ATOM_INPUT: layer_type_enum = LAYER_INPUT; break;
410 case ATOM_OUTPUT: layer_type_enum = LAYER_OUTPUT; break;
411 case ATOM_HIDDEN: layer_type_enum = LAYER_HIDDEN; break;
412 default: return error("Invalid layer type!\n");
413 }
414
415 if (layer_type_enum == LAYER_HIDDEN) {
416 const List* layer_size = p->next->next->next;
417 if (layer_size->data.type != VALUE_INT) {
418 return error("Hidden layer need int size.\n");
419 }
420 goetia_script_NewLayer(c, layer_name->data.atom, layer_type_enum, LAYER_VECTOR, layer_size->data.int32, NULL);
421 } else {
422 const List* layer_value = p->next->next->next;
423 if (layer_value->data.type != VALUE_ATOM) {
424 return error("Function 'lobe make' value type not.\n");
425 }
426
427 switch (layer_value->data.atom) {
428 case ATOM_CLASS:
429 const List* layer_classes = p->next->next->next->next;
430 if (layer_classes->data.type != VALUE_LIST) {
431 return error("Classes need to be list.\n");
432 }
433
434 layer_classes = layer_classes->data.list;
435
436 atom classes[10];
437 int class_count = 0;
438 for (int i = 0; i < 10; i++) {
439 if (!layer_classes) break;
440
441 if (layer_classes->data.type != VALUE_ATOM) {
442 return error("Classes need to be atom.\n");
443 }
444
445 classes[i] = layer_classes->data.atom;
446 class_count++;
447
448 layer_classes = layer_classes->next;
449 }
450
451 if (!class_count) {
452 return error("Classes need some classes in there\n");
453 }
454
455 goetia_script_NewLayer(c, layer_name->data.atom, layer_type_enum, LAYER_CLASS, class_count, classes);
456 break;
457 case ATOM_SCALAR:
458 goetia_script_NewLayer(c, layer_name->data.atom, layer_type_enum, LAYER_SCALAR, 1, NULL);
459 break;
460 case ATOM_VECTOR:
461 const List* layer_size = p->next->next->next;
462 if (layer_size->data.type != VALUE_INT) {
463 return error("Hidden layer need int size.\n");
464 }
465 goetia_script_NewLayer(c, layer_name->data.atom, layer_type_enum, LAYER_VECTOR, layer_size->data.int32, NULL);
466 break;
467 default: return error("Invalid layer value!\n");
468 }
469 }
470 break;
471
472 case ATOM_WEIGHT:
473 if (!p->next || !p->next->next || !p->next->next->next || !p->next->next->next->next || !p->next->next->next->next->next) {
474 return error("Function 'lobe weight' missing parameters.\n");
475 }
476
477 List* from_layer = p->next;
478 List* from_index = p->next->next;
479 List* to_layer = p->next->next->next;
480 List* to_index = p->next->next->next->next;
481 List* weight = p->next->next->next->next->next;
482
483 if (from_layer->data.type != VALUE_ATOM ||
484 from_index->data.type != VALUE_INT ||
485 to_layer->data.type != VALUE_ATOM ||
486 to_index->data.type != VALUE_INT ||
487 weight->data.type != VALUE_FLOAT
488 ) {
489 return error("Function 'lobe weight' type error.\n");
490 }
491
492 goetia_script_SetLayerWeight(c,
493 from_layer->data.atom,
494 from_index->data.int32,
495 to_layer->data.atom,
496 to_index->data.int32,
497 weight->data.float32
498 );
499 break;
500
501 case ATOM_SET: {
502 if (!p->next || !p->next->next || !p->next->next->next) {
503 return error("Function 'lobe set' missing parameters.\n");
504 }
505
506 List* layer = p->next;
507 List* index = p->next->next;
508 List* weight = p->next->next->next;
509
510 if (layer->data.type != VALUE_ATOM ||
511 index->data.type != VALUE_INT ||
512 weight->data.type != VALUE_FLOAT
513 ) {
514 return error("Function 'lobe set' type error.\n");
515 }
516
517 goetia_script_SetLayerInput(c,
518 layer->data.atom,
519 index->data.int32,
520 weight->data.float32
521 );
522 } break;
523 case ATOM_GET: {
524 if (!p->next || !p->next->next) {
525 return error("Function 'lobe get' missing parameters.\n");
526 }
527
528 List* layer = p->next;
529 List* index = p->next->next;
530
531 if (layer->data.type != VALUE_ATOM ||
532 index->data.type != VALUE_INT

539 index->data.int32
540 );
541
