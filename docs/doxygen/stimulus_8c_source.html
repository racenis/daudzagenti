83void goetia_stimulus_Process() {
84 int stimulus = stimulus_first;
85 stimulus_first = stimulus_last;
86
87 while (stimulus != stimulus_last) {
88 for (Agent* it = goetia_world_BeginAgent(); it; it = goetia_world_NextAgent(it)) {
89 if (!it->stimulus_script) continue;
90
91 // here we should check if the agent can receive the stimulus
92 // TODO: implement checking function
93
94 bind_stim_vals(it->context, &stimulus_queue[stimulus]);
95 goetia_script_Evaluate(it->context, it->stimulus_script);
96 }
97
98 stimulus++;
99 if (stimulus >= 100) stimulus = 0;
100 }
101}
