101Agent goetia_agent_AgentFromList(List* list) {
102 Agent agent = goetia_agent_New();
103
104 List* prop = list->data.list;
105
106 do switch (prop->data.atom) {
107 case ATOM_ID:
108 agent.id = prop->next->data.int32; break;
109 case ATOM_NAME:
110 agent.name = prop->next->data.atom; break;
111 case ATOM_DESCRIPTION:
112 agent.description = prop->next->data.str; break;
113 case ATOM_SPEED:
114 if (prop && (prop = prop->next)) agent.move_speed = get_float(prop);
115 if (prop && (prop = prop->next)) agent.turn_speed = get_float(prop);
116 if (prop && (prop = prop->next)) agent.accl_speed = get_float(prop);
117 break;
118 case ATOM_POSITION:
119 agent.position.x = get_float(prop->next);
120 agent.position.y = get_float(prop->next->next);
121 agent.position.z = get_float(prop->next->next->next);
122 break;
123 case ATOM_ROTATION:
124 agent.rotation = get_float(prop->next);
125 case ATOM_COLLISION:
126 switch (prop->next->data.atom) {
127 case ATOM_SPHERE:
128 agent.collision.type = SHAPE_SPHERE;
129 agent.collision.radius = get_float(prop->next->next);
130 break;
131 case ATOM_CYLINDER:
132 agent.collision.type = SHAPE_CYLINDER;
133 agent.collision.radius = get_float(prop->next->next);
134 break;
135 case ATOM_CUBE:
136 agent.collision.type = SHAPE_CUBE;
137 agent.collision.radius = get_float(prop->next->next);
138 break;
139 default: break;
140 } break;
141 case ATOM_MASS:
142 agent.mass = get_float(prop->next); break;
143 case ATOM_CHEMICAL:
144 agent.chemicals = goetia_chemical_ListToContainer(prop->next->data.list); break;
145 case ATOM_SKELETON: break;
146 case ATOM_CONTEXT: break; // TODO: implement
147
148 case ATOM_FLAGS:
149 while (prop = prop->next) switch (prop->data.atom) {
150 case ATOM_FLOAT_IN_AIR: agent.flags |= FLAG_FLOAT_IN_AIR; break;
151 case ATOM_FLOAT_IN_WATER: agent.flags |= FLAG_FLOAT_IN_WATER; break;
152 case ATOM_ALL_ENERGY: agent.flags |= FLAG_ALL_ENERGY; break;
153 case ATOM_ALL_SENSOR: agent.flags |= FLAG_ALL_SENSOR; break;
154 case ATOM_IMMORTAL: agent.flags |= FLAG_IMMORTAL; break;
155 case ATOM_DISALLOW_PICKUP: agent.flags |= FLAG_DISALLOW_PICKUP; break;
156 case ATOM_DYNAMICS_ASLEEP: agent.flags |= FLAG_DYNAMICS_ASLEEP; break;
157 case ATOM_DYNAMICS_DISABLE: agent.flags |= FLAG_DYNAMICS_DISABLE; break;
158 default:
159 printf("Unrecognized agent flag: %s.\n", goetia_atom_ToString(prop->data.atom));
160 } break;
161 break;
162
163 // scripts
164 case ATOM_INIT:
165 agent.init_script = prop->next->data.list; break;
166 case ATOM_UNINIT:
167 agent.uninit_script = prop->next->data.list; break;
168 case ATOM_TICK:
169 agent.tick_script = prop->next->data.list; break;
170 case ATOM_STIMULUS:
171 agent.stimulus_script = prop->next->data.list; break;
172 case ATOM_CATEGORY:
173 agent.category = prop->next->data.atom;
174 agent.category_visual = prop->next->next->data.atom;
175 agent.category_tactile = prop->next->next->next->data.atom;
176 case ATOM_ACTION:
177 agent.action_script = prop->next->data.list; break;
178
179
180 default:
181 goetia_PrintList(prop); printf(" | %i\n", prop->data.atom);
182 } while ((list = list->next) && (prop = list->data.list));
183
184 // bind default values
185 Value self_id_int;
186 self_id_int.type = VALUE_INT;
187 self_id_int.int32 = agent.id;
188
189 goetia_script_BindValue(agent.context, ATOM_AGENT_SELF, self_id_int);
190
191 return agent;
192}

200static bool collision(Agent* a, Agent* b) {
201 if (!a || !b) return false;
202
203 float distance = goetia_vec3_dst(a->position, b->position);
204
205 return distance < a->collision.radius + b->collision.radius;

208// turns agent toward target and moves toward it by dist
209static void move_toward(Agent* agent, Agent* target, float dist) {
210 Vector3 dir_raw = goetia_vec3_sub(target->position, agent->position);
211 Vector3 dir_nrm = goetia_vec3_nrm(dir_raw);
212 // change the speed to whatever
213 Vector3 dir_scl = goetia_vec3_mul(dir_nrm, goetia_vec3_scl(dist));
214 Vector3 new_pos = goetia_vec3_add(dir_scl, agent->position);
215
216 agent->position = new_pos;
217
218 Vector3 dir_flt = dir_raw;
219 dir_flt.y = 0.0f;
220 dir_flt = goetia_vec3_nrm(dir_flt);
221
222 agent->rotation = atan2f(-dir_flt.x, -dir_flt.z);
223
224 goetia_api_FireCallbackAgentMoved(agent->id, agent->position);
225 goetia_api_FireCallbackAgentRotated(agent->id, agent->rotation);
226
227 agent->flags &= ~FLAG_DYNAMICS_ASLEEP;
228}

233void goetia_agent_PerformAction(Agent* agent) {
234 Agent* target = goetia_world_FindAgent(agent->action.target);
235
236 // for some actions the agent needs to be in contact with target, so move it
237 // into contact for those actions.
238 switch (agent->action.type) {
239 case ACTION_NONE:
240 return;
241 case ACTION_PUSH:
242 case ACTION_PULL:
243 case ACTION_DEACTIVATE:
244 case ACTION_GET:
245 case ACTION_EAT:
246 case ACTION_HIT:
247 if (!target) {
248 printf("Agent target not found; cancelling.\n");
249 agent->action.type = ACTION_NONE;
250 return;
251 }
252
253 if (!collision(agent, target)) {
254 move_toward(agent, target, agent->move_speed);
255 return;
256 }
257
258 break;
259 case ACTION_DROP:
260 case ACTION_LOOKAT:
261 case ACTION_REST:
262 case ACTION_APPROACH:
263 case ACTION_RETREAT:
264 case ACTION_LEFT:
265 case ACTION_RIGHT:
266 case ACTION_UP:
267 case ACTION_DOWN:
268 case ACTION_FORWARD:
269 case ACTION_BACKWARD:
270 default: break;
271 }
272
273 // perform hard-coded actions
274 switch (agent->action.type) {
275 case ACTION_PUSH:
276 case ACTION_PULL:
277 case ACTION_DEACTIVATE:
278 case ACTION_EAT:
279 case ACTION_HIT:
280 case ACTION_LOOKAT:
281 case ACTION_REST:
282 case ACTION_APPROACH:
283 case ACTION_RETREAT:
284 default:
285 printf("action %s not implemented yet\n",
286 goetia_atom_ToString(goetia_action_ActionToAtom(agent->action.type)));
287 agent->action.type = ACTION_NONE;
288 return;
289
290 case ACTION_LEFT:
291 agent->rotation += agent->turn_speed;
292 goetia_api_FireCallbackAgentRotated(agent->id, agent->rotation);
293 return;
294 case ACTION_RIGHT:
295 agent->rotation -= agent->turn_speed;
296 goetia_api_FireCallbackAgentRotated(agent->id, agent->rotation);
297 return;
298 case ACTION_UP:
299 case ACTION_DOWN:
300 case ACTION_FORWARD:
301 case ACTION_BACKWARD:
302 Vector3 mov_dir;
303
304 switch (agent->action.type) {
305 case ACTION_UP:
306 mov_dir.x = 0.0f;
307 mov_dir.y = 1.0f;
308 mov_dir.z = 0.0f;
309 break;
310 case ACTION_DOWN:
311 mov_dir.x = 0.0f;
312 mov_dir.y = -1.0f;
313 mov_dir.z = 0.0f;
314 break;
315 case ACTION_FORWARD:
316 mov_dir.x = -sin(agent->rotation);
317 mov_dir.y = 0.0f;
318 mov_dir.z = -cos(agent->rotation);
319 break;
320 case ACTION_BACKWARD:
321 mov_dir.x = sin(agent->rotation);
322 mov_dir.y = 0.0f;
323 mov_dir.z = cos(agent->rotation);
324 break;
325 }
326
327 mov_dir = goetia_vec3_mul(mov_dir, goetia_vec3_scl(agent->move_speed));
328
329 agent->position = goetia_vec3_add(agent->position, mov_dir);
330
331 goetia_api_FireCallbackAgentMoved(agent->id, agent->position);
332 goetia_api_FireCallbackAgentRotated(agent->id, agent->rotation);
333
334 agent->flags &= ~FLAG_DYNAMICS_ASLEEP;
335
336 return;
337 case ACTION_GET:
338
339 // if agent is holding something already, drop it
340 if (agent->holding) {
341 goetia_world_FindAgent(agent->holding)->held_by = 0;
342 agent->holding = 0;
343 }
344
345 // pick up the target
346 agent->holding = agent->action.target;
347 target->held_by = agent->id;
348
349 return;
350 case ACTION_DROP:
351 if (agent->holding) {
352 goetia_world_FindAgent(agent->holding)->held_by = 0;
353 agent->holding = 0;
354 }
355 return;
356 }
357
358 // for scriptable actions; run their code
359
360 // run the action script on the target
361 if (target->action_script) {
362 if (agent->holding) {
363 goetia_script_BindValue(target->context, ATOM_AGENT_OBJECT, goetia_value_MakeInt(agent->holding));
364 } else {
365 goetia_script_BindValue(target->context, ATOM_AGENT_OBJECT, goetia_value_MakeAtom(ATOM_NULL));
366 }
367
368 // bind _action
369 goetia_script_BindValue(target->context, ATOM_AGENT_ACTION, goetia_value_MakeAtom(goetia_action_ActionToAtom(agent->action.type)));
370
371 // bind _subject
372 goetia_script_BindValue(target->context, ATOM_AGENT_SUBJECT, goetia_value_MakeInt(agent->id));
373
374 // evaluate target action script
375 goetia_script_Evaluate(target->context, target->action_script);
376 }
377
378 // action finished; remove action
379 agent->action.type = ACTION_NONE;
380 agent->action.target = AGENT_NONE;
381}
