265static Value arithmetic_compare(Context* c, List* expr, int sign) {
266 Value prev = unlist(c, expr);
267
268 while (expr = expr->next) {
269 Value next = unlist(c, expr);
270
271 int comp = compare(prev, next);
272
273 if (comp == -2) return error("Can't compare %s and %s!",
274 goetia_value_TypeToString(prev.type),
275 goetia_value_TypeToString(next.type)
276 );
277
278 if (comp != sign) return false;
279
280 prev = next;
281 }
282
283 return true;
284}

287static Value logic_operation(Context* c, List* expr, int op) {
288 do {
289 Value next_val = unlist(c, expr);
290 if (!goetia_value_Compare(next_val, true) &&
291 !goetia_value_Compare(next_val, false)
292 ) error("Invalid logic expr!\n");
293
294 int next = goetia_value_Compare(next_val, true);
295
296 if (op == -1) {
297 if (next) {
298 return false;
299 } else {
300 return true;
301 }
302 }
303
304 if (op == 0 && next) return true;
305 if (op == 1 && !next) return false;
306 } while (expr = expr->next);
307
308 if (op == -1) error("Mothing to not!\n");
309 if (op == 0) return false;
310 if (op == 1) return true;
311}

564Value apply(Context* context, Value func, List* expr) {
565 // need to function eval in case something is bound to it
566 func = eval(context, func);
567
568 // otherwise we assume that it is a built-in function
569 if (func.type != VALUE_ATOM) return error("Can only apply atom!\n");
570
571 switch (func.atom) {
572 case ATOM_ADD: return addition(context, expr);
573 case ATOM_SUB: return subtraction(context, expr);
574 case ATOM_MULT: return multiplication(context, expr);
575 case ATOM_DIV: return division(context, expr);
576 case ATOM_EQ: return arithmetic_compare(context, expr, 0);
577 case ATOM_GT: return arithmetic_compare(context, expr, 1);
578 case ATOM_LT: return arithmetic_compare(context, expr, -1);
579 case ATOM_AND: return logic_operation(context, expr, 1);
580 case ATOM_OR: return logic_operation(context, expr, 0);
581 case ATOM_NOT: return logic_operation(context, expr, -1);
582 case ATOM_PRINT: return print(context, expr);
583 case ATOM_IF: return if_expr(context, expr);
584 case ATOM_SET: return set(context, expr);
585 case ATOM_QUOTE: return quote(context, expr);
586 case ATOM_LIST: return list(context, expr);
587 case ATOM_EVAL: return eval_func(context, expr);
588 case ATOM_APPLY: return apply_func(context, expr);
589 case ATOM_CONTEXT: print_context(context, expr); return ok;
590 case ATOM_DO: return eval_list(context, expr);
591 case ATOM_PATTERN: return pattern(context, expr);
592 case ATOM_IS: return is(context, expr);
593
594 default: break;
595 }
596
597 // this file is getting kinda long, let's continue in another one
598 return goetia_script_SimulationFunction(context, func.atom, expr);
599}

601Value eval(Context* context, Value expr) {
602 // if evaluation an atom, first check if it is bound to something
603 if (expr.type == VALUE_ATOM) {
604 int bind = find_bind(context, expr.atom);
605
606 if (bind == -1) {
607 return expr;
608 } else {
609 return context->values[bind].value;
610 }
611 }
612
613 // otherwise, if not list, then return expression as-is
614 if (expr.type != VALUE_LIST) return expr;
615
616 // if list has only a single element, then evaluate just that element
617 // I don't remember why?
618 if (expr.list->next == NULL) return eval(context, expr.list->data);
619
620 // find the function
621 Value func = eval(context, expr.list->data);
622
623 // apply function to its parameters
624 return apply(context, func, expr.list->next);
625}
