79Sector goetia_world_SectorFromList(List* list) {
80 Sector sector = goetia_world_MakeSector();
81 List* prop = list->data.list;
82
83 sector.flags = 0;
84
85 do switch (prop->data.atom) {
86 case ATOM_NAME:
87 sector.name = prop->next->data.atom; break;
88 case ATOM_WALL:
89 if (prop->next->data.atom == ATOM_TRUE) sector.flags |= SECTOR_SOLID_WALL_1;
90 if (prop->next->next->data.atom == ATOM_TRUE) sector.flags |= SECTOR_SOLID_WALL_2;
91 if (prop->next->next->next->data.atom == ATOM_TRUE) sector.flags |= SECTOR_SOLID_WALL_3;
92 if (prop->next->next->next->next->data.atom == ATOM_TRUE) sector.flags |= SECTOR_SOLID_WALL_4;
93 if (prop->next->next->next->next->next->data.atom == ATOM_TRUE) sector.flags |= SECTOR_SOLID_CEILING;
94 if (prop->next->next->next->next->next->next->data.atom == ATOM_TRUE) sector.flags |= SECTOR_SOLID_FLOOR;
95 break;
96 case ATOM_X:
97 sector.x[0] = get_float(prop->next->data);
98 sector.x[1] = get_float(prop->next->next->data);
99 sector.x[2] = get_float(prop->next->next->next->data);
100 sector.x[3] = get_float(prop->next->next->next->next->data);
101 break;
102 case ATOM_Z:
103 sector.z[0] = get_float(prop->next->data);
104 sector.z[1] = get_float(prop->next->next->data);
105 sector.z[2] = get_float(prop->next->next->next->data);
106 sector.z[3] = get_float(prop->next->next->next->next->data);
107 break;
108 case ATOM_Y_MAX:
109 sector.y_max[0] = get_float(prop->next->data);
110 sector.y_max[1] = get_float(prop->next->next->data);
111 sector.y_max[2] = get_float(prop->next->next->next->data);
112 sector.y_max[3] = get_float(prop->next->next->next->next->data);
113 break;
114 case ATOM_Y_MIN:
115 sector.y_min[0] = get_float(prop->next->data);
116 sector.y_min[1] = get_float(prop->next->next->data);
117 sector.y_min[2] = get_float(prop->next->next->next->data);
118 sector.y_min[3] = get_float(prop->next->next->next->next->data);
119 break;
120 case ATOM_FLAGS:
121 while (prop = prop->next) switch (prop->data.atom) {
122 case ATOM_AIR: sector.flags |= SECTOR_AIR; break;
123 case ATOM_WATER: sector.flags |= SECTOR_WATER; break;
124 default:
125 printf("Unrecognized sector flag: %s.\n", goetia_atom_ToString(prop->data.atom));
126 } break;
127 break;
128 default:
129 goetia_PrintList(prop); printf(" | %i\n", prop->data.atom);
130 } while ((list = list->next) && (prop = list->data.list));
131
132 if (sector.flags & SECTOR_AIR) sector.chemicals->properties &= ~CHEM_ALLOW_SOLUBLE;
133
134 return sector;
135}

142static Plane get_plane(Sector* sec, int p) {
143 Vector3 a, b, c;
144 switch(p) {
145 case 0:
146 case 1:
147 case 2:
148 case 3:
149 a.x = sec->x[p];
150 a.y = sec->y_min[p];
151 a.z = sec->z[p];
152 b.x = sec->x[p];
153 b.y = sec->y_max[p];
154 b.z = sec->z[p];
155 c.x = sec->x[(p+1)%4];
156 c.y = sec->y_min[(p+1)%4];
157 c.z = sec->z[(p+1)%4];
158 break;
159 case 4:
160 a.x = sec->x[0];
161 a.y = sec->y_max[0];
162 a.z = sec->z[0];
163 b.x = sec->x[1];
164 b.y = sec->y_max[1];
165 b.z = sec->z[1];
166 c.x = sec->x[2];
167 c.y = sec->y_max[2];
168 c.z = sec->z[2];
169 break;
170 case 5:
171 a.x = sec->x[0];
172 a.y = sec->y_min[0];
173 a.z = sec->z[0];
174 b.x = sec->x[2];
175 b.y = sec->y_min[2];
176 b.z = sec->z[2];
177 c.x = sec->x[1];
178 c.y = sec->y_min[1];
179 c.z = sec->z[1];
180 }
181
182 Vector3 ab = goetia_vec3_sub(b, a);
183 Vector3 ac = goetia_vec3_sub(c, a);
184 Vector3 cr = goetia_vec3_nrm(goetia_vec3_crs(ab, ac));
185
186 float d = -(cr.x*a.x + cr.y*a.y + cr.z*a.z);
187
188 Plane plane;
189 plane.x = -cr.x;
190 plane.y = -cr.y;
191 plane.z = -cr.z;
192 plane.w = -d;
193
194 return plane;
195}

218sector_id goetia_world_InsertSector(Sector sector) {
219 if (sector_count >= MAX_SECTORS) {
220 printf("Sector insertion aborted; MAX_SECTORS exceeded.\n");
221 return 0;
222 }
223
224 sector.id = sector_count;
225
226 // this bit here will check if there is any sector adjacent to this one,
227 // and if there are such sectors, they will be connected together.
228
229 for (int sec = 1; sec < sector_count; sec++) {
230 int points_in_this = 0;
231
232 for (int point = 0; point < 8; point++) {
233
234 int fail = 0;
235 for (int plane = 0; plane < 6; plane++) {
236 Plane test_plane = get_plane(&sectors[sec], plane);
237 Vector3 test_point = get_point(&sector, point);
238
239 if (plane_dist(test_plane, test_point) < 0.0f) {
240 fail++;
241 }
242 }
243
244 if (!fail) {
245 points_in_this++;
246 }
247 }
248
249 if (points_in_this >= 4) {
250 // TODO: make this bit a bit nicer
251 int inserted = 0;
252 for (int i = 0; i < 6; i++) {
253 if (sector.connections[i]) continue;
254
255 sector.connections[i] = sectors[sec].id;
256 inserted = 1;
257 break;
258 }
259
260 if (!inserted) {
261 printf("Failed to connect, maximum connections!\n");
262 }
263
264 inserted = 0;
265 for (int i = 0; i < 6; i++) {
266 if (sectors[sec].connections[i]) continue;
267
268 sectors[sec].connections[i] = sector.id;
269 inserted = 1;
270 break;
271 }
272
273 if (!inserted) {
274 printf("Failed to connect, maximum connections!\n");
275 }
276 }
277 }
278
279 sectors[sector_count] = sector;
280
281 return sector_count++;
282}

444static void apply_physics(Agent* agent) {
445 if (agent->flags & (FLAG_DYNAMICS_ASLEEP | FLAG_DYNAMICS_DISABLE)) return;
446
447 Vector3 initial_pos = agent->position;
448 sector_id sector = 0;
449
450 // add gravity
451 agent->velocity.y -= 0.01f;
452 agent->position = goetia_vec3_add(agent->position, agent->velocity);
453
454 for (Agent* other = goetia_world_BeginAgent(); other; other = goetia_world_NextAgent(other)) {
455 if (agent == other) continue;
456
457 // this only does spherical collisions
458 // TODO: implement other collisions properly
459
460 float distance = goetia_vec3_dst(agent->position, other->position);
461 float min_dist = agent->collision.radius + other->collision.radius;
462
463 if (distance < min_dist) {
464 // push agent out of the other one!
465 float penetration = min_dist - distance;
466
467 Vector3 push_dir = goetia_vec3_sub(agent->position, other->position);
468 push_dir = goetia_vec3_nrm(push_dir);
469 push_dir = goetia_vec3_mul(push_dir, goetia_vec3_scl(penetration));
470
471 agent->position = goetia_vec3_add(agent->position, push_dir);
472
473 Stimulus collision;
474 collision.type = 0;
475
476 collision.type = TOUCH_AGENT;
477 //collision.sector = sec->id;
478 collision.sector = 0;
479 collision.position = other->position;
480
481 collision.subject = agent->id;
482 collision.object = other->id;
483 collision.action = 0;
484
485 goetia_stimulus_Emit(collision);
486 }
487
488 }
489
490 // this will be filled in if the agent wallbonks
491 Stimulus wallbonk;
492 wallbonk.type = 0;
493
494 // iterate through all sectors.
495 // in the future it would be a good idea to use a BVH or something to narrow
496 // down possible collidable sectors, instead of iterating all of them and
497 // checking for collisions with each one.
498 for (Sector* sec = goetia_world_BeginSector(); sec; sec = goetia_world_NextSector(sec)) {
499
500 // check if agent's center is inside of this sector
501 for (int p = 0; p < 6; p++) {
502 Plane plane = get_plane(sec, p);
503
504 if (plane_dist(plane, agent->position) < 0.0f) {
505 goto next; // this fails the test
506 }
507 }
508
509 sector = sec->id;
510
511 // check collisions for all walls and push the agent if necessary
512 for (int p = 0; p < 6; p++) {
513 switch (p) {
514 case 0:
515 if (!(sec->flags & SECTOR_SOLID_WALL_1)) continue; break;
516 case 1:
517 if (!(sec->flags & SECTOR_SOLID_WALL_2)) continue; break;
518 case 2:
519 if (!(sec->flags & SECTOR_SOLID_WALL_3)) continue; break;
520 case 3:
521 if (!(sec->flags & SECTOR_SOLID_WALL_4)) continue; break;
522 case 4:
523 if (!(sec->flags & SECTOR_SOLID_CEILING)) continue; break;
524 case 5:
525 if (!(sec->flags & SECTOR_SOLID_FLOOR)) continue; break;
526 }
527
528
529 Plane plane = get_plane(sec, p);
530 float dist = plane_dist(plane, agent->position);
531
532 // this collision check only works for spheres
533 // TODO: add checks for all other collision shapes
534 if (dist < agent->collision.radius) {
535 float push = agent->collision.radius - dist;
536 Vector3 dir;
537 dir.x = plane.x;
538 dir.y = plane.y;
539 dir.z = plane.z;
540 dir = goetia_vec3_mul(dir, goetia_vec3_scl(push));
541 agent->position = goetia_vec3_add(agent->position, dir);
542 agent->velocity.y = 0.0f;
543
544 // TODO: add code to differentiate different bonk stimuli.
545 // like wallbonk/ceiling bonk/floor bonk
546
547 wallbonk.type = TOUCH_WALLBONK;
548 wallbonk.sector = sec->id;
549 wallbonk.position = agent->position;
550
551 wallbonk.subject = agent->id;
552 wallbonk.object = 0;
553 wallbonk.action = 0;
554 }
555 }
556
557 break;
558next:
559 }
560
561 // check if agent moved to a different sector
562 if (sector && agent->sector != sector) {
563
564 // fire the 'leave' stimulus, if agent was inside of a sector
565 if (agent->sector) {
566 Stimulus leave;
567
568 leave.type = VISUAL_LEAVE;
569 leave.sector = agent->sector;
570 leave.position = initial_pos;
571
572 leave.subject = agent->id;
573 leave.object = 0;
574 leave.action = 0;
575
576 goetia_stimulus_Emit(leave);
577 }
578
579 // fire the 'appear' stimulus, if agent has entered a new sector
580 if (sector) {
581 Stimulus appear;
582
583 appear.type = VISUAL_APPEAR;
584 appear.sector = agent->sector;
585 appear.position = agent->position;
586
587 appear.subject = agent->id;
588 appear.object = 0;
589 appear.action = 0;
590
591 goetia_stimulus_Emit(appear);
592 }
593
594 // set the new sector
595 agent->sector = sector;
596 }
597
598 // fire callback and stimulus if the agent moved
599 if (initial_pos.x != agent->position.x ||
600 initial_pos.y != agent->position.y ||
601 initial_pos.z != agent->position.z
602 ) {
603 Stimulus move;
604
605 move.type = VISUAL_MOVE;
606 move.sector = agent->sector;
607 move.position = agent->position;
608
609 move.subject = agent->id;
610 move.object = 0;
611 move.action = 0;
612
613 goetia_stimulus_Emit(move);
614
615 if (wallbonk.type) {
616 goetia_stimulus_Emit(wallbonk);
617 }
618
619 goetia_api_FireCallbackAgentMoved(agent->id, agent->position);
620 } else {
621 agent->flags |= FLAG_DYNAMICS_ASLEEP;
622 //printf("fell asleep %i \n", agent->id);
623 }
624}

662void goetia_world_Tick() {
663 agent_id expired = AGENT_NONE;
664 for (Agent* it = agents; it != &agents[agent_count]; it++) {
665 if (it->flags & FLAG_DEAD) {
666 expired = it->id;
667 continue;
668 }
669
670 if (it->tick_script) {
671 goetia_script_Evaluate(it->context, it->tick_script);
672 }
673
674 if (it->chemicals) {
675 goetia_chemical_UpdateContainer(it->chemicals);
676 }
677
678 // perform agent's action if it has enough chemical energy, or if it has
679 // the free energy flag.
680 // TODO: put this in a seperate function
681 if (it->flags & FLAG_ALL_ENERGY) {
682 goetia_agent_PerformAction(it);
683 } else if (it->chemicals && it->chemicals->chems[CHEM_ATP]) {
684 it->chemicals->chems[CHEM_ATP]--;
685 it->chemicals->chems[CHEM_ADP]++;
686
687 goetia_agent_PerformAction(it);
688 }
689
690
691 apply_physics(it);
692 }
693
694 if (expired) {
695 printf("Agent got expired!\n");
696 goetia_world_RemoveAgent(expired);
697 }
698
699
700 // we should add some sort of a diffusion rate parameter to the chemical
701 // diffusion function
702
703 // this rate should also be calculated from the size of the shared
704 // connection area
705
706 for (Sector* it = sectors; it != &sectors[sector_count]; it++) {
707 // diffuse chemicals
708 for (int i = 0; i < 6; i++) {
709 if (!it->connections[i]) continue;
710
711 goetia_chemical_Diffuse(it->chemicals, sectors[it->connections[i]].chemicals);
712 }
713 }
