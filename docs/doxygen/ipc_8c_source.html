38static bool set_nonblocking(int socket) {
39#ifdef _WIN32
40 unsigned long non_zero = 1;
41 return (ioctlsocket(socket, FIONBIO, &non_zero) == 0);
42#else
43 int flags = fcntl(socket, F_GETFL, 0);
44 if (flags == -1) return false;
45 flags = flags | O_NONBLOCK;
46 return (fcntl(socket, F_SETFL, flags) == 0);
47#endif
48}

51void goetia_ipc_StartServer() {
52 int error;
53#ifdef _WIN32
54 WSADATA wsa_data;
55 error = WSAStartup(MAKEWORD(2, 2), &wsa_data);
56 if (error != 0) {
57 printf("Winsock startup failed: %i\n", error);
58 abort();
59 }
60#endif
61
62 memset(&server_info, 0, sizeof(server_info));
63
64 server_info.sin_family = AF_INET;
65 server_info.sin_port = htons(4444); // change port to whatever
66 server_info.sin_addr.s_addr = inet_addr("127.0.0.1");
67
68 server_socket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
69 if (server_socket == INVALID_SOCKET) {
70 printf("Error opening server socket: %i\n", ERROR_VALUE);
71 abort();
72 }
73
74 error = bind(server_socket, (struct sockaddr*)&server_info, sizeof(server_info));
75 if (error == SOCKET_ERROR) {
76 printf("Error binding server socket: %i\n", ERROR_VALUE);
77 abort();
78 }
79
80 if (!set_nonblocking(server_socket)) {
81 printf("error setting nonblcoking socket\n");
82 abort();
83 }
84
85 error = listen(server_socket, 1);
86 if (error == SOCKET_ERROR) {
87 printf("Error listening server socket: %i\n", ERROR_VALUE);
88 abort();
89 }
90}
