169int goetia_chemical_React(ChemContainer* container, chem chem1, chem chem2, chem prod) {
170 const Reaction* r = NULL;
171 for (int i = 0; i < enzyme_reaction_count; i++) {
172 if (enzyme_reactions[i].reactant1 == chem1 &&
173 enzyme_reactions[i].reactant2 == chem2 &&
174 enzyme_reactions[i].product == prod
175 ) {
176 r = &enzyme_reactions[i];
177 break;
178 }
179
180 if (enzyme_reactions[i].reactant1 == chem2 &&
181 enzyme_reactions[i].reactant2 == chem1 &&
182 enzyme_reactions[i].product == prod
183 ) {
184 r = &enzyme_reactions[i];
185
186 chem chem_swap = chem1;
187 chem1 = chem2;
188 chem2 = chem_swap;
189
190 break;
191 }
192 }
193
194 // invalid reaction
195 if (!r) return 0;
196
197 // no catalyst
198 if (r->catalyst && !container->chems[r->catalyst]) return 0;
199
200 // not enough reactants
201 if (r->reactant1_ammount > container->chems[r->reactant1] ||
202 r->reactant2_ammount > container->chems[r->reactant2]
203 ) return 0;
204
205 container->chems[r->reactant1] -= r->reactant1_ammount;
206 container->chems[r->reactant2] -= r->reactant2_ammount;
207
208 container->chems[r->product] += r->product_ammount;
209 container->chems[r->byproduct] += r->byproduct_ammount;
210
211 return r->product_ammount;
212}

268void goetia_chemical_Diffuse(ChemContainer* c1, ChemContainer* c2) {
269 if (!(c1->properties & CHEM_ALLOW_DIFFUSION) ||
270 !(c2->properties & CHEM_ALLOW_DIFFUSION)) {
271 return;
272 }
273
274 for (chem chem = 1; chem < CHEM_LAST; chem++) {
275 if (((goetia_chemical_Property(chem) & CHEM_SOLUBLE) &&
276 (c2->properties & CHEM_ALLOW_SOLUBLE)) || ((goetia_chemical_Property(chem) & CHEM_VOLATILE) &&
277 (c2->properties & CHEM_ALLOW_VOLATILE))
278 ) {
279 if (c1->chems[chem] > c2->chems[chem]) {
280 c1->chems[chem]--;
281 c2->chems[chem]++;
282 }
283 }
284 }
